NameClickHouse Storage Usage per Device Group
DescriptionProvides storage used by each Top level CMDB Device Group, e.g. Network, Servers etc
Data SourceFortiSIEM
Result TypeQuery
SQL Query
        
WITH
    total_disk_size AS
    (
        SELECT
            hostName() AS host,
            sum(bytes_on_disk) AS total_bytes
        FROM cluster('fsiem_cluster', system.parts)
        WHERE (table = 'events_replicated') AND active
        GROUP BY ALL
    ),
    perGroup AS
    (
        SELECT
            host,
            device_group,
            message_bytes_one_group,
            sum(message_bytes_one_group) OVER (PARTITION BY host) AS total_message_bytes,
            event_count
        FROM
        (
            SELECT
                hostName() AS host,
                multiIf(dictHas('fsiem.dict_cmdb_group', (phCustId, if(position(reptDevIpAddr, ':') = 0, concat('::ffff:', reptDevIpAddr), reptDevIpAddr), 'PH_SYS_DEVICE_Network:IntfIP')), 'Network', dictHas('fsiem.dict_cmdb_group', (phCustId, if(position(reptDevIpAddr, ':') = 0, concat('::ffff:', reptDevIpAddr), reptDevIpAddr), 'PH_SYS_DEVICE_Server:IntfIP')), 'Server', dictHas('fsiem.dict_cmdb_group', (phCustId, if(position(reptDevIpAddr, ':') = 0, concat('::ffff:', reptDevIpAddr), reptDevIpAddr), 'PH_SYS_DEVICE_CLOUD_SERVICE:IntfIP')), 'Cloud Service', 'Others') AS device_group,
                sumMerge(`SUM(rawEventSize)`) AS message_bytes_one_group,
                countMerge(`COUNT(*)`) AS event_count
            FROM cluster('fsiem_cluster', fsiem.mv_events)
            GROUP BY ALL
        )
    )
SELECT
    device_group,
    toUInt64(SUM((message_bytes_one_group / total_message_bytes) * total_bytes)) AS bytes_on_disk,
    sum(event_count) AS total_events
FROM perGroup
INNER JOIN total_disk_size USING (host)
GROUP BY ALL
ORDER BY bytes_on_disk DESC

NameDecline in events from a reporting device
DescriptionIdentify all devices that have reported 50% less events in the past hour compared to the average of the previous four hours. Devices that send less than 10 events in the hour are excluded.
Data SourceFortiSIEM
Result TypeQuery
SQL Query
SELECT * EXCEPT hour
FROM
(
    SELECT
        reptDevName,
        reptDevIpAddrV4 as reptDevIP,
        reptVendor,
        reptModel,
        toStartOfHour(phRecvTime) AS hour,
        count() AS count,
        round(avg(count) OVER (PARTITION BY (reptDevName,reptDevIP,reptVendor,reptModel) ORDER BY hour ASC ROWS BETWEEN 4 PRECEDING AND 1 PRECEDING)) AS avg_count_prev_4_hours,
        round(count / avg_count_prev_4_hours, 2) AS ratio
    FROM fsiem.events
    WHERE (phRecvTime >= toStartOfHour(now() - (5 * 3600))) AND (phRecvTime < toStartOfHour(now()))
    GROUP BY
        reptDevName,
        reptDevIP,
        reptVendor,
        reptModel,
        hour
)
WHERE (hour >= toStartOfHour(now() - 3600)) AND (ratio < 0.5) AND (count > 10)
ORDER BY ratio DESC

NameFortiADC: WAF Signature Events with URL containing GTFOBins
DescriptionDetects that a FortiADC WAF Signature log contains an URL referring to a a GTFOBin process. GTFOBin (https://gtfobins.github.io/) is a curated list of trusted Unix binaries that can be used to bypass local security restrictions in misconfigured systems. This query requires that you import data into the GTFOBin Lookup Table by creating CSV file from https://gtfobins.github.io/ and also import into the (already created) LOLBAS Lookup Table in FortiSIEM.
Data SourceFortiADC via Syslog
Result TypeQuery
SQL Query
        
WITH
    arrayDistinct(arrayFlatten(groupArray(arrayDistinct(splitByChar(';', dict.functions))))) AS unique_functions
SELECT
    e.phRecvTime,
    e.reptDevIpAddrV4,
    e.reptDevName,
    e.eventType,
    e.infoURL,
    e.decoded_url,
    arrayStringConcat(unique_functions, '; ') AS functions,
    arrayStringConcat(arrayDistinct(groupArray(dict.lolbin)), ', ') AS matched_command
FROM
(
    SELECT
        phRecvTime,
        reptDevIpAddrV4,
        reptDevName,
        eventType,
        metrics_string.value[indexOf(metrics_string.name, 'infoURL')] AS infoURL,
        decodeURLComponent(metrics_string.value[indexOf(metrics_string.name, 'infoURL')]) AS decoded_url
    FROM fsiem.events
    WHERE
        indexOf(metrics_string.name, 'infoURL') > 0
        AND eventType = 'FortiADC-Security-WAF-Signature'
        AND phRecvTime > now() - INTERVAL 7 DAY
) AS e
CROSS JOIN fsiem.GTFOBin_0 AS dict
WHERE
    match(lower(e.decoded_url), CONCAT('(^|\\s)', lower(dict.lolbin), '(\\s|$)'))
    OR match(lower(e.infoURL), CONCAT('(^|\\s)', lower(dict.lolbin), '(\\s|$)'))
GROUP BY
    e.phRecvTime,
    e.reptDevIpAddrV4,
    e.reptDevName,
    e.eventType,
    e.infoURL,
    e.decoded_url;

NameWindows: Command Line Obfuscation using escape characters
DescriptionDetects command line parameter obfuscation by matching for 10 or more occurrences of dots, backslashes, dollar signs, spaces, percentage signs, plus sign or percent signs.
Data SourceMicrosoft Windows Security Log via OMI or FortiSIEM Agent, Microsoft Windows Sysmon via FortiSIEM Agent
Result TypeQuery
SQL Query
        
SELECT
    phRecvTime,
    reptDevIpAddrV4,
    reptDevName,
    eventType,
    metrics_string.value[indexOf(metrics_string.name, 'parentProcName')] AS parent,
    procName,
    metrics_string.value[indexOf(metrics_string.name, 'command')] AS command,
    (length(command) - length(replaceRegexpAll(command, '[^\\x00-\\x7F]', ''))) > 0 AS hasUnicode,
countMatches(command, '"[a-zA-Z]"') > 0 AS hasSplitQuoting,
(length(command) - length(lower(command)) > 0 AND length(command) - length(upper(command)) > 0) AS hasMixedCase,
countMatches(command, '\\.') > 15 AS excessiveDots,
countMatches(command, '\\\\') > 10 AS excessiveBackslashes,
countMatches(command, '\\$') > 11 AS excessiveDollarSigns,
countMatches(command, '\\%') > 10 AS excessivePercentSigns,
countMatches(command, '\\^') > 10 AS excessiveCarets,
countMatches(command, '\\+') > 10 AS excessivePluses,
countMatches(command, '\\s') > 10 AS excessiveSpaces,
countMatches(command, '\\"') > 10 AS excessiveQuotes,
(
 positionCaseInsensitive(command, 'char') > 0 OR
 positionCaseInsensitive(command, '[char]') > 0 OR 
 positionCaseInsensitive(command, 'chr(') > 0 OR
 positionCaseInsensitive(command, 'fromCharCode') > 0
) AS containsChar,
countMatches(command, '''') > 10 AS excessiveSingleQuotes,

(
  toInt8((length(command) - length(replaceRegexpAll(command, '[^\\x00-\\x7F]', ''))) > 0) * 3 +
  toInt8(countMatches(command, '"[a-zA-Z]"') > 0) * 3 +
  toInt8((length(command) - length(lower(command)) > 0 AND length(command) - length(upper(command)) > 0)) * 2 +

  toInt8(countMatches(command, '\\\\') > 10) +

  toInt8(countMatches(command, '\\+') > 10) +
  toInt8(countMatches(command, '\\s') > 10) +
  toInt8(countMatches(command, '\\"') > 10) +
  toInt8(
    positionCaseInsensitive(command, 'char') > 0 OR
    positionCaseInsensitive(command, '[char]') > 0 OR 
    positionCaseInsensitive(command, 'chr(') > 0 OR
    positionCaseInsensitive(command, 'fromCharCode') > 0
  ) * 2 +
  toInt8(countMatches(command, '''') > 10)
) AS huntScore
FROM fsiem.events
WHERE (eventType IN ('Win-Sysmon-1-Create-Process', 'Win-Security-4688')) 
AND ((metrics_string.value[indexOf(metrics_string.name, 'command')]) IS NOT NULL)

AND (NOT (parent regexp '(?i)(manage\s?engine|chrome|firefox|edge|adobe|forticlient)'))

GROUP BY
    phRecvTime,
    reptDevIpAddrV4,
    reptDevName,
    eventType,
    parent,
    procName,
    command
HAVING huntScore >= 1
ORDER BY huntScore DESC

NameWindows: LOLBIN Process Creation
DescriptionDetects processes that exactly match the process names specified in the LOLBAS (Living Off The Land Binaries, Scripts and Libraries) project https://lolbas-project.github.io. These processes are trusted system tools and are not malicious themselves, but can be weaponized maliciously by cyber attackers to avoid detection. More details can be found here https://www.fortinet.com/resources/cyberglossary/living-off-the-land-lotl. This query requires that you download the CSV file from https://lolbas-project.github.io/api/ and import into (already created) FortiSIEM LOLBAS Lookup Table.
Data SourceMicrosoft Windows Security Log via OMI or FortiSIEM Agent, Microsoft Windows Sysmon via FortiSIEM Agent
Result TypeQuery
SQL Query
        
WITH
    (SELECT groupArray(lolbin) FROM fsiem.LOLBAS_0)                    AS lolbins,
    arrayMap(x -> concat('(?i)\\b', x, '\\b'), lolbins)               AS patterns

/* ------------------------------------------------------------------ */
/* 2.  Pull events for the period, extract the “command” string once  */
/* ------------------------------------------------------------------ */
SELECT
    ev.phRecvTime,
    ev.reptDevIpAddrV4,
    ev.reptDevName,
    ev.eventType,
    ev.cmd                                               AS command,

    /* tags and matched lolbins, deduplicated & concatenated */
    arrayStringConcat(groupArrayDistinct(d.tags), ', ')  AS tagName,
    arrayStringConcat(groupArrayDistinct(match_name), ', ') AS commandMatch
FROM
(
    SELECT
        phRecvTime,
        reptDevIpAddrV4,
        reptDevName,
        eventType,

        /* one expression to pull the command */
        metrics_string.value[
            indexOf(metrics_string.name, 'command')
        ]                                                AS cmd
    FROM fsiem.events
    WHERE  eventType IN ('Win-Sysmon-1-Create-Process', 'Win-Security-4688')
      AND  phRecvTime >= now() - INTERVAL 7 DAY
      AND  has(metrics_string.name, 'command')           -- guarantees index > 0
) AS ev

/* ------------------------------------------------------------------ */
/* 3.  Explode **only** the patterns that matched this command        */
/* ------------------------------------------------------------------ */
ARRAY JOIN
    arrayMap(i -> lolbins[i],
             multiMatchAllIndices(ev.cmd, patterns))     AS match_name

/* ------------------------------------------------------------------ */
/* 4.  Bring in the LOLBAS metadata for each match                    */
/* ------------------------------------------------------------------ */
LEFT JOIN fsiem.LOLBAS_0 AS d
       ON d.lolbin = match_name

GROUP BY
    ev.phRecvTime,
    ev.reptDevIpAddrV4,
    ev.reptDevName,
    ev.eventType,
    ev.cmd

NameWindows: LOLBIN Process launched from unexpected path
DescriptionDetects processes that exactly match the process names specified in the LOLBAS (Living Off The Land Binaries, Scripts and Libraries) project is being launched from a location that does not match the definition in the LOLBAS project. LOLBAS project is found here: https://lolbas-project.github.io. These LOLBAS processes (also called LOLBINs) are trusted system tools and are not malicious themselves, but can be weaponized maliciously by cyber attackers to avoid detection. This query requires that you import data into the (already created) FortiSIEM LOLBAS Lookup Table by downloading the CSV file from https://lolbas-project.github.io/api/.
Data SourceMicrosoft Windows Security Log via OMI or FortiSIEM Agent, Microsoft Windows Sysmon via FortiSIEM Agent
Result TypeQuery
SQL Query
        
SELECT
    phRecvTime,
    reptDevIpAddrV4,
    reptDevName,
    eventType,
    metrics_string.value[indexOf(metrics_string.name, 'command')] AS command,
    procName,
    arrayStringConcat(groupArrayDistinct(dict.tags), ', ') AS tagName,
    arrayStringConcat(groupArrayDistinct(dict.lolbin), ', ') AS matched_command,
    arrayStringConcat(groupArrayDistinct(dict.paths), ', ') AS expected_paths,
    if(arrayExists(path -> procName ILIKE CONCAT('%', path, '%'), splitByString(';', arrayStringConcat(groupArrayDistinct(dict.paths), ', '))), 'Path Match', 'Path Mismatch') AS path_match_status
FROM fsiem.events
CROSS JOIN fsiem.LOLBAS_0 AS dict
WHERE
    indexOf(metrics_string.name, 'command') > 0
    AND metrics_string.value[indexOf(metrics_string.name, 'command')] ILIKE CONCAT('%', dict.lolbin, '%')
    AND eventType IN ('Win-Security-4688', 'Win-Sysmon-1-Create-Process')
    AND phRecvTime > now() - INTERVAL 7 DAY
GROUP BY phRecvTime, reptDevIpAddrV4, reptDevName, eventType, command, procName
HAVING path_match_status = 'Path Mismatch'
	



NameWindows: LOLBIN Process with command matching LOTS website
DescriptionDetects processes that exactly match the process names specified in the LOLBAS (Living Off The Land Binaries, Scripts and Libraries) project https://lolbas-project.github.io, and the command line parameters matches the websites defined in the LOTS (Living Off Trusted Sites) Project https://lots-project.com/. These LOLBAS processes (also called LOLBINs) are trusted system tools and are not malicious themselves, but can be weaponized maliciously by cyber attackers to avoid detection. The LOTS project contains popular legitimate domains that are used by attackers to conduct phishing, Command and Control, exfiltration and downloading tools to evade detection. This query has 2 requirements. You need to download the CSV file from LOLBAS website: https://lolbas-project.github.io/api/ and import into the (already created) FortiSIEM LOLBAS Lookup Table. Additionally, you need to create CSV file from LOTS website: https://lots-project.com/ and import into the (already created) FortiSIEM LOTS Lookup Table.
Data SourceMicrosoft Windows Security Log via OMI or FortiSIEM Agent, Microsoft Windows Sysmon via FortiSIEM Agent
Result TypeQuery
SQL Query
        
SELECT
    e.phRecvTime,
    e.reptDevIpAddrV4,
    e.reptDevName,
    e.eventType,
    e.command AS CommandLine,
    e.shortProcName,
    l.lolbin AS LOLBAS_Match,
    lots.websites AS LOTS_Match
FROM
    (
        SELECT
            phRecvTime,
            reptDevIpAddrV4,
            reptDevName,
            eventType,
            metrics_string.value[indexOf(metrics_string.name, 'command')] AS command,

        FROM fsiem.events

    AND e.command ILIKE CONCAT('%', lots.websites, '%');

NameWindows: Long Duration Terminal Services Sessions
DescriptionTracks Windows Terminal Service Sessions that last for more than 1 hour.
Data SourceMicrosoft Windows Terminal Services LocalSessionManager/Operational logs via FortiSIEM Agent
Result TypeQuery
SQL Query
        
WITH
    session_start_events AS (
        SELECT
            reptDevIpAddrV4 AS "Reporting IP",
            reptDevName AS "Reporting Device",
            user AS "User Name",
            srcIpAddrV4 AS "Source IP",
            metrics_string.value[indexOf(metrics_string.name,'sessionId')] AS "Session ID",
            phRecvTime AS "Session Start Time",
            row_number() OVER (
                PARTITION BY user,
                             metrics_string.value[indexOf(metrics_string.name,'sessionId')]
                ORDER BY phRecvTime
            ) AS sessionIndex
        FROM fsiem.events
        WHERE eventType IN (
            'Win-TerminalServices-LocalSessionManager-21',
            'Win-TerminalServices-LocalSessionManager-25'
        )
        AND phRecvTime >= now() - INTERVAL 14 DAY
    ),

    session_end_events AS (
        SELECT
            reptDevIpAddrV4 AS "Reporting IP",
            reptDevName AS "Reporting Device",
            user AS "User Name",
            NULLIF(srcIpAddrV4,'') AS "Source IP",
            metrics_string.value[indexOf(metrics_string.name,'sessionId')] AS "Session ID",
            phRecvTime AS "Session End Time",
            row_number() OVER (
                PARTITION BY user,
                             metrics_string.value[indexOf(metrics_string.name,'sessionId')]
                ORDER BY phRecvTime
            ) AS sessionEndIndex
        FROM fsiem.events
        WHERE eventType IN (
            'Win-TerminalServices-LocalSessionManager-24',
            'Win-TerminalServices-LocalSessionManager-23'
        )
        AND phRecvTime >= now() - INTERVAL 14 DAY
    )

SELECT *
FROM (
    SELECT
        ss."Reporting IP",
        ss."Reporting Device",
        ss."User Name",
        ss."Source IP" AS "Start Source IP",
        ss."Session ID",
        ss."Session Start Time" AS "Logon/Reconnection Time",
        se."Session End Time" AS "Disconnect/Logoff Time",

        multiIf(formatDateTime(ss."Session Start Time",'%w') = '0','Sunday',
                formatDateTime(ss."Session Start Time",'%w') = '1','Monday',
                formatDateTime(ss."Session Start Time",'%w') = '2','Tuesday',
                formatDateTime(ss."Session Start Time",'%w') = '3','Wednesday',
                formatDateTime(ss."Session Start Time",'%w') = '4','Thursday',
                formatDateTime(ss."Session Start Time",'%w') = '5','Friday',
                'Saturday') AS "Logon Day",

        multiIf(formatDateTime(se."Session End Time",'%w') = '0','Sunday',
                formatDateTime(se."Session End Time",'%w') = '1','Monday',
                formatDateTime(se."Session End Time",'%w') = '2','Tuesday',
                formatDateTime(se."Session End Time",'%w') = '3','Wednesday',
                formatDateTime(se."Session End Time",'%w') = '4','Thursday',
                formatDateTime(se."Session End Time",'%w') = '5','Friday',
                'Saturday') AS "Logout Day",

        greatest(se."Session End Time" - ss."Session Start Time", 0) AS "Duration Seconds",

        concat(
            toString(intDiv(greatest(se."Session End Time" - ss."Session Start Time", 0), 3600)), 'h ',
            toString(intDiv(greatest(se."Session End Time" - ss."Session Start Time", 0) % 3600, 60)), 'm ',
            toString(greatest(se."Session End Time" - ss."Session Start Time", 0) % 60), 's'
        ) AS "Session Duration",

        multiIf(
            (toHour(ss."Session Start Time") BETWEEN 8 AND 17)
            AND formatDateTime(ss."Session Start Time",'%w') IN ('1','2','3','4','5'),
            'Business Hours',
            'Non-Business Hours'
        ) AS "Business Logon Hours"
    FROM session_start_events ss
    JOIN session_end_events se
        ON ss."User Name" = se."User Name"
        AND ss."Source IP" = se."Source IP"
        AND ss."Session ID" = se."Session ID"
        AND ss.sessionIndex = se.sessionEndIndex
        AND ss."Reporting Device" = se."Reporting Device"

    UNION ALL

    SELECT
        ss."Reporting IP",
        ss."Reporting Device",
        ss."User Name",
        ss."Source IP" AS "Start Source IP",
        ss."Session ID",
        ss."Session Start Time" AS "Logon/Reconnection Time",
        se."Session End Time" AS "Disconnect/Logoff Time",

        multiIf(formatDateTime(ss."Session Start Time",'%w') = '0','Sunday',
                formatDateTime(ss."Session Start Time",'%w') = '1','Monday',
                formatDateTime(ss."Session Start Time",'%w') = '2','Tuesday',
                formatDateTime(ss."Session Start Time",'%w') = '3','Wednesday',
                formatDateTime(ss."Session Start Time",'%w') = '4','Thursday',
                formatDateTime(ss."Session Start Time",'%w') = '5','Friday',
                'Saturday') AS "Logon Day",

        multiIf(formatDateTime(se."Session End Time",'%w') = '0','Sunday',
                formatDateTime(se."Session End Time",'%w') = '1','Monday',
                formatDateTime(se."Session End Time",'%w') = '2','Tuesday',
                formatDateTime(se."Session End Time",'%w') = '3','Wednesday',
                formatDateTime(se."Session End Time",'%w') = '4','Thursday',
                formatDateTime(se."Session End Time",'%w') = '5','Friday',
                'Saturday') AS "Logout Day",

        greatest(se."Session End Time" - ss."Session Start Time", 0) AS "Duration Seconds",

        concat(
            toString(intDiv(greatest(se."Session End Time" - ss."Session Start Time", 0), 3600)), 'h ',
            toString(intDiv(greatest(se."Session End Time" - ss."Session Start Time", 0) % 3600, 60)), 'm ',
            toString(greatest(se."Session End Time" - ss."Session Start Time", 0) % 60), 's'
        ) AS "Session Duration",

        multiIf(
            (toHour(ss."Session Start Time") BETWEEN 8 AND 17)
            AND formatDateTime(ss."Session Start Time",'%w') IN ('1','2','3','4','5'),
            'Business Hours',
            'Non-Business Hours'
        ) AS "Business Logon Hours"
    FROM session_start_events ss
    JOIN session_end_events se
        ON ss."User Name" = se."User Name"
        AND se."Source IP" IS NULL
        AND ss."Session ID" = se."Session ID"
        AND ss.sessionIndex = se.sessionEndIndex
        AND ss."Reporting Device" = se."Reporting Device"
) AS combined_results
WHERE "Duration Seconds" >= 3600
ORDER BY "Duration Seconds" DESC, "User Name";
	
      
NameWindows: Suspicious PowerShell Cmdlet Match
DescriptionDetects Windows Active Directory commands that those specified in the LOLAD (Living Off The Land and Exploitation Active Directory) project. Details can be found in https://lolad-project.github.io/. This is a collection of Active Directory techniques, commands, and functions that can be used natively to support offensive security operations and Red Team exercises. This query requires that you import data into the LOLAD Lookup Table by creating CSV file from https://lolad-project.github.io/ and then import into the (already created) LOLAD FortiSIEM Lookup Table.
Data SourceMicrosoft Windows Security Log via OMI or FortiSIEM Agent, Microsoft Windows Sysmon via FortiSIEM Agent
Result TypeQuery
SQL Query
        
SELECT
    phRecvTime,
    reptDevName,
    metrics_string.value[indexOf(metrics_string.name, 'script')] AS script,
    arrayStringConcat(groupArrayDistinct(dict.description), ', ') AS description,
    arrayStringConcat(groupArrayDistinct(dict.toolkit), ', ') AS toolkit,
    arrayStringConcat(groupArrayDistinct(dict.cmdlet), ', ') AS commandMatch
FROM fsiem.events
CROSS JOIN fsiem.MaliciousPowershellCmdlet_0 AS dict
WHERE (indexOf(metrics_string.name, 'script') > 0) AND (eventType IN ('Win-PowerShell-4104')) AND match(metrics_string.value[indexOf(metrics_string.name, 'script')], CONCAT('(?i)(^|\\s)', dict.cmdlet, '(\\s|$)')) AND (phRecvTime >= (now() - toIntervalDay(7)))
GROUP BY
    phRecvTime,
    reptDevName,
    script
