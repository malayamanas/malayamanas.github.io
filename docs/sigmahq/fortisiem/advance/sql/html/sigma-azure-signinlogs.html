-- ============================================================
-- Title:        Account Lockout
-- Sigma ID:     2b7d6fc0-71ac-4cf7-8ed1-b5788ee5257a
-- Level:        medium  |  FSM Severity: 5
-- Status:       test
-- MITRE:        T1110

-- Unmapped:     ResultType

-- ============================================================

SELECT
  phRecvTime,
  reptDevName,
  reptDevIpAddrV4,
  user,
  rawEventMsg
FROM fsiem.events
WHERE rawEventMsg LIKE '%'

LIMIT 1000;
